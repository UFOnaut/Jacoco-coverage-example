package com.illia.noteapp.utils

import com.illia.noteapp.Constants
import com.illia.noteapp.Constants.DATE_FORMAT
import com.illia.noteapp.Constants.UI_DATE_FORMAT
import java.text.SimpleDateFormat
import java.util.*
import java.util.concurrent.TimeUnit

object DateTimeUtils {

    @JvmStatic
    fun formatSavedDateForUI(timeStamp: String?): String? {
        if (timeStamp != null) {
            try {
                val formatter = SimpleDateFormat(DATE_FORMAT, Locale.US)
                val date = formatter.parse(timeStamp)
                val outputFormatter = SimpleDateFormat(UI_DATE_FORMAT, Locale.US)
                return outputFormatter.format(date)
            } catch (e: Exception) {}

        }
        return timeStamp
    }

    @JvmStatic
    fun getCurrentTimeStampForDb(): String? {
        val formatter = SimpleDateFormat(DATE_FORMAT, Locale.US)
        return formatter.format(System.currentTimeMillis())
    }
}
